import os
import speech_recognition as sr
import pyttsx3
import webbrowser
from google_auth_oauthlib.flow import InstalledAppFlow
from googleapiclient.discovery import build

# 1. Setup YouTube API with OAuth
SCOPES = ['https://www.googleapis.com/auth/youtube.force-ssl']

def get_authenticated_service():
    flow = InstalledAppFlow.from_client_secrets_file('client_secrets.json', SCOPES)
    credentials = flow.run_local_server(port=0)
    return build('youtube', 'v3', credentials=credentials)

# 2. Voice Assistant Setup
engine = pyttsx3.init()

def speak(text):
    engine.say(text)
    engine.runAndWait()

def listen():
    r = sr.Recognizer()
    with sr.Microphone() as source:
        print("Listening...")
        audio = r.listen(source)
        try:
            query = r.recognize_google(audio)
            print(f"User said: {query}")
            return query.lower()
        except:
            return ""

# 3. Search and Play Logic
def play_on_youtube(service, song_name):
    request = service.search().list(
        q=song_name,
        part='snippet',
        type='video',
        maxResults=1
    )
    response = request.execute()
    
    if response['items']:
        video_id = response['items'][0]['id']['videoId']
        url = f"https://www.youtube.com/watch?v={video_id}"
        speak(f"Playing {song_name} on YouTube.")
        webbrowser.open(url)
    else:
        speak("I couldn't find that song.")

# Main Execution
if __name__ == "__main__":
    youtube_service = get_authenticated_service()
    speak("Assistant ready. What song should I play?")
    
    while True:
        command = listen()
        if 'play' in command:
            song = command.replace('play', '').strip()
            play_on_youtube(youtube_service, song)
        elif 'stop' in command:
            break
